function doGet(e) {
	var urlData = e.parameter.url;
	var postData = e.parameter.post;
	
	if (urlData==undefined) {
		return ContentService.createTextOutput('Error no urlData. Params are url= and post=')
	}
	if (postData==undefined) {
		var options = {
			"method"  : "GET",   
			"followRedirects" : true,
			"muteHttpExceptions": true
		}
	}
	else {
		var options = {
			"method"  : "post",
			"contentType": "application/json",
			"payload" : postData
		}
	}
	var result = UrlFetchApp.fetch(urlData, options);
	
	return ContentService.createTextOutput(result);
}
